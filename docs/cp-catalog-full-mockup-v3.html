Transformation SQL (dbt model)
{{ config(materialized='table', tags=['client','eod']) }}
SELECT account_number,
       client_id,
       INITCAP(account_display_name)             AS client_name,
       portfolio_id,
       UPPER(base_currency)                      AS base_currency,
       market_value_local * fx_rate              AS market_value_usd,
       CASE WHEN aum_tier >= 3 THEN 'PRIORITY'
            ELSE 'STANDARD' END                  AS service_tier,
       CURRENT_DATE                              AS as_of_date
FROM {{ ref('sei_slv_client') }}
WHERE account_status = 'ACTIVE'
